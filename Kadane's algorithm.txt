class Solution{

    // arr: input array
    // n: size of array
    //Function to find the sum of contiguous subarray with maximum sum.
    long maxSubarraySum(int arr[], int n){
        
        long maxSum = Integer.MIN_VALUE,presentSum = 0;
        for(int i=0;i<n;i++){
            presentSum += arr[i];
            
            maxSum = Math.max(presentSum,maxSum);
            
            if(presentSum<0)
                presentSum = 0;
        }
        return maxSum;
    }
    
}
