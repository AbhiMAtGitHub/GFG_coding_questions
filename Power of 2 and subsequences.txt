class Solution{
    static Long numberOfSubsequences(int N, ArrayList<Long> A){
        long mod = 1000000007,count=0,ans=1;
        
        for(int i=0;i<N;i++){
            if((A.get(i) & A.get(i)-1)==0)
                count++;
        }
        
        while(count>0){
            ans = (ans*2)%mod;
            count--;
        }
        return ans-1;
    }
}
