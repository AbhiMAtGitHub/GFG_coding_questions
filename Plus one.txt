class Solution {
    static ArrayList<Integer> increment(ArrayList<Integer> arr , int N) {
        ArrayList<Integer> list = new ArrayList<>();
        int carry = 1, sum = 0;
        for(int i=N-1;i>=0;i--){
            list.add((arr.get(i)+carry)%10);
            carry = (carry+arr.get(i))/10;
        }
        if(carry!=0)
            list.add(carry);
        Collections.reverse(list);
        return list;
    }
};
