class Solution
{
    public String digitsNum(int N)
    {
        String res = "";
        
        if(N%9!=0)
            res = res + String.valueOf(N%9);
        
        for(int i=1;i<=(N/9);i++)
            res = res + "9";
            
        for(int i=0;i<N;i++)
            res = res + "0";
        
        return res;
    }
}
