class Solution {
    
    public void segregateElements(int arr[], int n)
    {
        int count = 0;
        int temp[] = new int[n];
        for(int i=0;i<n;i++){
            if(arr[i]>0)
                temp[count++] = arr[i];
        }
        
        for(int i=0;i<n;i++){
            if(arr[i]<0)
                temp[count++] = arr[i];
        }
        
        for(int i=0;i<n;i++){
            arr[i] = temp[i];
        }
    }
}
