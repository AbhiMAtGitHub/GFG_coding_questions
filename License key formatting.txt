class Solution{
    static String ReFormatString(String S, int K){
        int count = 0;
        StringBuilder res = new StringBuilder();
        for(int i=S.length()-1;i>=0;i--){
            char ch = S.charAt(i);
            if(ch=='-')
                continue;
            else if(Character.isLowerCase(ch)){
                res.append(Character.toUpperCase(ch));
            }
            else{
                res.append(ch);
            }
            count++;
            if(count==K){
                res.append('-');
                count = 0;
            }
        }
        res.reverse();
        if(res.length()>0 && res.charAt(0)=='-')
            res.deleteCharAt(0);
        return res.toString();
    }
}
