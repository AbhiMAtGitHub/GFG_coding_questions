// Return the sum of all the factors of n;

class Solution {
    static long maxSumLCM(int n) {
        long ans = 0;
        for(int i=1;i<=Math.sqrt(n);i++){
            if(n%i==0){
                ans+=i;
                if(n/i!=i)
                    ans+=(n/i);
            }
        }
        return ans;
    }
}
