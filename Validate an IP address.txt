def isValid(s):
    if ".." in s or s.count('.')>3 or s.count('.')<3:
        return False
    list1 = s.split(".")
    if len(list1)>4 or len(list1)<4:
        return False
    for i in list1:
        if i.isalpha():
            return False
        if i!=str(int(i)):
            return False
        if int(i) not in range(0,256):
            return False
    return True
